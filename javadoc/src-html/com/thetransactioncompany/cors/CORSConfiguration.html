001    package com.thetransactioncompany.cors;
002    
003    
004    import java.util.HashSet;
005    import java.util.Properties;
006    import java.util.Set;
007    
008    import com.thetransactioncompany.util.PropertyParseException;
009    import com.thetransactioncompany.util.PropertyRetriever;
010    
011    
012    /**
013     * The CORS filter configuration (typically originating from the web application
014     * descriptor file {@code web.xml}). The fields become immutable (final) after 
015     * the are initialised.
016     *
017     * @author Vladimir Dzhuvinov
018     * @version 1.3.2 (2011-07-29)
019     */
020    public class CORSConfiguration {
021            
022            
023            /**
024             * If {@code true} generic HTTP requests must be allowed to pass 
025             * through the filter, else only valid and accepted CORS requests must 
026             * be allowed (strict CORS filtering).
027             *
028             * <p>Property key: cors.allowGenericHttpRequests
029             */
030            public final boolean allowGenericHttpRequests;
031    
032    
033            /**
034             * If {@code true} the CORS filter must allow requests from any origin,
035             * else the origin whitelist {@link #allowedOrigins} must be consulted.
036             *
037             * <p>Property key: cors.allowOrigin (set to {@code *})
038             */
039            public final boolean allowAnyOrigin;
040            
041            /**
042             * If {@code true} the CORS filter must allow requests from any origin
043             * whose suffix matches any of the {@link #allowedOrigins}.
044             *
045             * <p>Property key: cors.allowOrigin (set to {@code *})
046             */
047            public final boolean allowOriginSuffixMatching;
048    
049    
050            /**
051             * Whitelisted origins that the CORS filter must allow. Requests from 
052             * origins not included here must be refused with a HTTP 403 "Forbidden"
053             * response. This property is overriden by {@link #allowAnyOrigin}.
054             *
055             * <p>Note: The set is of type String instead of Origin to bypass
056             * parsing of the request origins before matching, see 
057             * http://lists.w3.org/Archives/Public/public-webapps/2010JulSep/1046.html
058             *
059             * <p>Property key: cors.allowOrigin
060             */
061            public final Set<String> allowedOrigins;
062    
063            /**
064             * Helper method to check whether requests from the specified origin 
065             * must be allowed. This is done by looking up {@link #allowAnyOrigin} 
066             * and {@link #allowedOrigins} (in that order).
067             *
068             * @param origin The origin as reported by the web client (browser), 
069             *               {@code null} if unknown.
070             *
071             * @return {@code true} if the origin is allowed, else {@code false}.
072             */
073            public final boolean isAllowedOrigin(final String origin) {
074    
075            if (allowAnyOrigin)
076                return true;
077                    
078            if (this.allowOriginSuffixMatching)
079                    return originSuffixAllowed(origin);
080            
081                    if (origin == null)
082                            return false;
083                    
084            if (allowedOrigins.contains(origin))
085                            return true;
086                    else
087                            return false;
088            }
089    
090            /**
091             * Helper method to check whether requests from the specified origin suffix and scheme
092             * are allowed. This is done by looking up the Origin's scheme and host name 
093             * and matching with {@link #allowedOrigins} but also allowing for subdomains.
094             * eg. Origin: https://foo.example.com matches cors.allowedOrigin = https://example.com
095             * whereas cors.allowedOrigin = http://example.com would not match.
096             *
097             * @param originString The origin as reported by the web client (browser), 
098             *               {@code null} if unknown.
099             *
100             * @return {@code true} if the origin suffix is allowed, else {@code false}.
101             */
102            public final boolean originSuffixAllowed(final String originString) {
103                    
104                    boolean allowed = false;
105                    
106                    try {
107                            Origin origin = new Origin(originString);
108                            
109                            String originSuffix = origin.getSuffix();
110                            
111                            for (String allowedOriginString: allowedOrigins) {
112                                    Origin allowedOrigin = new Origin(allowedOriginString);
113                                    if (originSuffix.endsWith(allowedOrigin.getSuffix()) && origin.getScheme().equalsIgnoreCase(allowedOrigin.getScheme()))
114                                            return true;
115                            }
116                            
117                    } catch (OriginException e) {
118                            return allowed;
119                    }
120                    return allowed;
121            }
122            
123            /**
124             * The supported HTTP methods. Requests for methods not included here 
125             * must be refused by the CORS filter with a HTTP 405 "Method not 
126             * allowed" response.
127             *
128             * <p>Property key: cors.supportedMethods
129             */
130            public final Set<HTTPMethod> supportedMethods;
131            
132            
133            /**
134             * Helper method to check whether the specified HTTP method is 
135             * supported. This is done by looking up {@link #supportedMethods}.
136             *
137             * @param method The HTTP method.
138             *
139             * @return {@code true} if the method is supported, else {@code false}.
140             */
141            public final boolean isSupportedMethod(final HTTPMethod method) {
142            
143                    if (supportedMethods.contains(method))
144                            return true;
145                    else
146                            return false;
147            }
148            
149            
150            /**
151             * The names of the supported author request headers.
152             * 
153             * <p>Property key: cors.supportedHeaders
154             */
155            public final Set<HeaderFieldName> supportedHeaders;
156            
157            
158            /**
159             * Helper method to check whether the specified (non-simple) author 
160             * request header is supported. This is done by looking up 
161             * {@link #supportedHeaders}.
162             *
163             * @param header The header field name.
164             *
165             * @return {@code true} if the header is supported, else {@code false}.
166             */
167            public final boolean isSupportedHeader(final HeaderFieldName header) {
168    
169                    if (supportedHeaders.contains(header))
170                            return true;
171                    else
172                            return false;
173            }
174            
175            
176            /**
177             * The non-simple response headers that the web browser should expose to 
178             * the author of the CORS request.
179             *
180             * <p>Property key: cors.exposedHeaders
181             */
182            public final Set<HeaderFieldName> exposedHeaders;
183            
184            
185            /**
186             * Indicates whether user credentials, such as cookies, HTTP 
187             * authentication or client-side certificates, are supported.
188             *
189             * <p>Property key: cors.supportsCredentials
190             */
191            public final boolean supportsCredentials;
192            
193            
194            /**
195             * Indicates how long the results of a preflight request can be cached
196             * by the web client, in seconds. If {@code -1} unspecified.
197             *
198             * <p>Property key: cors.maxAge
199             */
200            public final int maxAge;
201            
202            
203            /**
204             * Parses a string containing words separated by space and/or comma.
205             *
206             * @param s The string to parse.
207             *
208             * @return An array of the parsed words, empty if none were found.
209             */
210            protected static String[] parseWords(final String s) {
211                    
212                    String s1 = s.trim();
213                    
214                    if (s1.isEmpty())
215                            return new String[]{};
216                    else
217                            return s1.split("\\s*,\\s*|\\s+");
218            }
219            
220            
221            /**
222             * Creates a new CORS configuration from the specified properties.
223             *
224             * <p>The following properties are recognized (if missing they default
225             * to the specified values):
226             *
227             * <ul>
228             *     <li>cors.allowGenericHttpRequests {true|false} defaults to {@code true}.
229             *     <li>cors.allowOriginSuffixMatching {true|false} defaults to {@code false}.
230             *     <li>cors.allowOrigin {"*"|origin-list} defaults to {@code *}.
231             *     <li>cors.supportedMethods {method-list} defaults to {@code "GET, POST, HEAD, OPTIONS"}.
232             *     <li>cors.supportedHeaders {header-list} defaults to empty list.
233             *     <li>cors.exposedHeaders {header-list} defaults to empty list.
234             *     <li>cors.supportsCredentials {true|false} defaults to {@code true}.
235             *     <li>cors.maxAge {int} defaults to {@code -1} (unspecified).
236             * </ul>
237             *
238             * @param props The properties.
239             *
240             * @throws CORSConfigurationException On a invalid property.
241             */
242            public CORSConfiguration(final Properties props)
243                    throws CORSConfigurationException {
244            
245                    try {
246                            PropertyRetriever pr = new PropertyRetriever(props);
247    
248                            // Parse the allow generic HTTP requests option
249                            
250                            allowGenericHttpRequests = pr.getOptBoolean("cors.allowGenericHttpRequests", true);
251    
252                            
253                            allowOriginSuffixMatching = pr.getOptBoolean("cors.allowOriginSuffixMatching", false);
254    
255                            // Parse the allowed origins list
256                            
257                            String originSpec = pr.getOptString("cors.allowOrigin", "*").trim();
258                            
259                            allowedOrigins = new HashSet<String>();
260                            
261                            if (originSpec.equals("*")) {
262    
263                                    allowAnyOrigin = true;
264                            }
265                            else {
266    
267                                    allowAnyOrigin = false;
268    
269                                    String[] urls = parseWords(originSpec);
270    
271                                    for (String url: urls) {
272    
273                                            try {
274                                                    
275                                                    Origin origin = new Origin(url);
276                                                    allowedOrigins.add(origin.toString());
277    
278                            } catch (OriginException e) {
279                                    throw new PropertyParseException("Bad origin URL in property cors.allowOrigin: " + url);
280                            }
281                                    }
282                            }
283                            
284    
285                            // Parse the supported methods list
286    
287                            String methodSpec = pr.getOptString("cors.supportedMethods", "GET, POST, HEAD, OPTIONS").trim().toUpperCase();
288    
289                            String[] methodNames = parseWords(methodSpec);
290    
291                            supportedMethods = new HashSet<HTTPMethod>();
292    
293                            for (String methodName: methodNames) {
294    
295                                    try {
296                                            supportedMethods.add(HTTPMethod.valueOf(methodName));
297    
298                                    } catch (IllegalArgumentException e) {
299                                            throw new PropertyParseException("Bad HTTP method name in property cors.allowMethods: " + methodName);
300                                    }
301                            }
302                            
303    
304                            // Parse the supported headers list
305    
306                            String[] headers = parseWords(pr.getOptString("cors.supportedHeaders", ""));
307    
308                            supportedHeaders = new HashSet<HeaderFieldName>();
309    
310                            for (String header: headers) {
311    
312                                    try {
313                                            supportedHeaders.add(new HeaderFieldName(header));
314    
315                                    } catch (IllegalArgumentException e) {
316                                            throw new PropertyParseException("Bad header field name in property cors.supportedHeaders: " + header);
317                                    }
318                            }
319    
320    
321                            // Parse the exposed headers list
322                            
323                            headers = parseWords(pr.getOptString("cors.exposedHeaders", ""));
324    
325                            exposedHeaders = new HashSet<HeaderFieldName>();
326    
327                            for (String header: headers) {
328    
329                                    try {
330                                            exposedHeaders.add(new HeaderFieldName(header));
331    
332                                    } catch (IllegalArgumentException e) {
333                                            throw new PropertyParseException("Bad header field name in property cors.exposedHeaders: " + header);
334                                    }
335                            }
336    
337    
338                            // Parse the allow credentials option
339                            supportsCredentials = pr.getOptBoolean("cors.supportsCredentials", true);
340    
341    
342                            // Parse the max cache age of preflight requests
343                            maxAge = pr.getOptInt("cors.maxAge", -1);
344                            
345                    
346                    } catch (PropertyParseException e) {
347                            
348                            // Simply reformat as a more specific exception class
349                            // to improve stack trace clarity (config exceptions
350                            // are often dumped to the web client screen)
351                            
352                            throw new CORSConfigurationException(e.getMessage());
353                    }
354            }
355    }
