001    package com.thetransactioncompany.cors;
002    
003    
004    /**
005     * Unsupported HTTP header exception. Thrown to indicate that a custom HTTP
006     * request header is not supported by the CORS policy.
007     *
008     * @author Vladimir Dzhuvinov
009     * @version 1.3.2 (2010-09-27)
010     */
011    public class UnsupportedHTTPHeaderException extends CORSException {
012    
013            
014            /**
015             * The HTTP header.
016             */
017            private HeaderFieldName header = null;
018            
019            
020            /**
021             * Creates a new unsupported HTTP header exception with the specified 
022             * message.
023             *
024             * @param message The message.
025             */
026            public UnsupportedHTTPHeaderException(final String message) {
027            
028                    this(message, null);
029            }
030            
031            
032            /**
033             * Creates a new unsupported HTTP header exception with the specified 
034             * message and request header.
035             *
036             * @param message       The message.
037             * @param requestHeader The request HTTP header name, {@code null} if 
038             *                      unknown.
039             */
040            public UnsupportedHTTPHeaderException(final String message, final HeaderFieldName requestHeader) {
041            
042                    super(message);
043                    
044                    header = requestHeader;
045            }
046            
047            
048            /**
049             * Gets the request header name.
050             *
051             * @return The request header name, {@code null} if unknown or not set.
052             */
053            public HeaderFieldName getRequestHeader() {
054            
055                    return header;
056            }
057    }
