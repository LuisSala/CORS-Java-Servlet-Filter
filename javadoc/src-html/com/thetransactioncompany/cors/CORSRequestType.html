001    package com.thetransactioncompany.cors;
002    
003    
004    import javax.servlet.http.HttpServletRequest;
005    
006    
007    /**
008     * Enumeration of the CORS request types.
009     *
010     * @author Vladimir Dzhuvinov
011     * @version 1.3.2 (2011-12-02)
012     */
013    public enum CORSRequestType {
014    
015            
016            /**
017             * Simple / actual CORS request.
018             */
019            ACTUAL,
020            
021            
022            /**
023             * Preflight CORS request.
024             */
025            PREFLIGHT,
026            
027            
028            /**
029             * Other (non-CORS) request.
030             */
031            OTHER;
032            
033            
034            /**
035             * Detects the CORS type of the specified HTTP request.
036             *
037             * @param request The HTTP request to check. Must not be {@code null}.
038             *
039             * @return The CORS request type.
040             */
041            public static CORSRequestType detect(final HttpServletRequest request) {
042            
043                    if (request == null)
044                            throw new NullPointerException("The HTTP request must not be null");
045                    
046                    // All CORS request have an Origin header
047                    if (request.getHeader("Origin") == null)
048                            return OTHER;
049                    
050                    // We have a CORS request - determine type
051                    if (request.getHeader("Access-Control-Request-Method") != null &&
052                        request.getMethod()                                != null &&
053                        request.getMethod().equals("OPTIONS")                         )
054                        
055                            return PREFLIGHT;
056                            
057                    else
058                            return ACTUAL;
059            }
060    }
